Click the Circle

The Function moves the circle randomly on the body content when the mouse hovers over it.

Named Function

            const circle = document.querySelector(".circle"); 
            const body = document.querySelector('body');

            circle.addEventListener('mouseover', function(){
                const x = Math.floor(Math.random() * (body.clientWidth - 100));
                const y = Math.floor(Math.random() * (body.clientHeight - 100));

                circle.style.left = `${x}px`;
                circle.style.top = `${y}px`;
            });

Function Expression

            const circle = document.querySelector(".circle"); 
            const body = document.querySelector('body');

            const moveCircle = function(){
                const x = Math.floor(Math.random() * (body.clientWidth - 100));
                const y = Math.floor(Math.random() * (body.clientHeight - 100));

                circle.style.left = `${x}px`;
                circle.style.top = `${y}px`;
            };

            //pass into event
            circle.addEventListener('mouseover', moveCircle);
